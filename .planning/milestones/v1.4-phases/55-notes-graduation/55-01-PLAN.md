```python
class TagAction(CommandModel):
    __doc__ = TAG_ACTION_DOC
    add: Patch[list[str]] = Field(default=UNSET, description=TAG_ACTION_ADD)
    remove: Patch[list[str]] = Field(default=UNSET, description=TAG_ACTION_REMOVE)
    replace: PatchOrClear[list[str]] = Field(default=UNSET, description=TAG_ACTION_REPLACE)

    @model_validator(mode="after")
    def _validate_incompatible_tag_edit_modes(self) -> TagAction:
        has_replace = is_set(self.replace)
        has_add = is_set(self.add)
        has_remove = is_set(self.remove)
        if has_replace and (has_add or has_remove):
            raise ValueError(TAG_REPLACE_WITH_ADD_REMOVE)
        if not has_replace and not has_add and not has_remove:
            raise ValueError(TAG_NO_OPERATION)
        return self
```
