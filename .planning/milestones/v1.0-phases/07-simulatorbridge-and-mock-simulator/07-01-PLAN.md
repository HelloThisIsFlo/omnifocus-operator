```python
def create_bridge(bridge_type: str) -> Bridge:
    """bridge_type: One of "inmemory", "simulator", or "real"."""
    match bridge_type:
        case "simulator":
            msg = "SimulatorBridge not yet implemented (Phase 7)"
            raise NotImplementedError(msg)
        case "real":
            import os
            from omnifocus_operator.bridge._real import DEFAULT_IPC_DIR, RealBridge
            ipc_dir_str = os.environ.get("OMNIFOCUS_IPC_DIR")
            ipc_dir = Path(ipc_dir_str) if ipc_dir_str else DEFAULT_IPC_DIR
            return RealBridge(ipc_dir=ipc_dir)
```

```python
async def app_lifespan(app: FastMCP) -> AsyncIterator[dict[str, object]]:

    bridge = create_bridge(bridge_type)
    if hasattr(bridge, "ipc_dir"):
        await sweep_orphaned_files(bridge.ipc_dir)
```
