       def _normalize_for_comparison(payload: dict) -> dict:
           """Strip volatile fields (IDs, timestamps) so the snapshot is stable across runs.

           Mirrors the existing golden-master normalization pattern used elsewhere in
           tests/golden_master (if present — check and reuse).

           # Deep-normalize: IDs, added/modified timestamps, URL.
           volatile_keys = {"id", "added", "modified", "url"}
           def scrub(obj):
               if isinstance(obj, dict):
                   return {k: scrub(v) if k not in volatile_keys else "<normalized>" for k, v in obj.items()}
               if isinstance(obj, list):
                   return [scrub(x) for x in obj]
               return obj
           return scrub(payload)
