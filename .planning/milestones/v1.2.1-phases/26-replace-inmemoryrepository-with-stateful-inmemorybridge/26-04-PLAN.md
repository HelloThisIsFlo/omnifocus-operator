**Add fixtures to `tests/conftest.py`** (append after existing factory functions):

```python
import pytest
from omnifocus_operator.repository import BridgeRepository
from omnifocus_operator.service import OperatorService
from tests.doubles import ConstantMtimeSource, InMemoryBridge

@pytest.fixture
def bridge(request: pytest.FixtureRequest) -> InMemoryBridge:
    """InMemoryBridge seeded from @pytest.mark.snapshot(...) or default snapshot.

    Usage:
        @pytest.mark.snapshot(tasks=[make_task_dict(id="t1")])
        async def test_something(self, service): ...

    Without marker: uses make_snapshot_dict() defaults (1 of each entity).
    """
    marker = request.node.get_closest_marker("snapshot")
    if marker is not None:
        snapshot_data = make_snapshot_dict(**marker.kwargs)
    else:
        snapshot_data = make_snapshot_dict()
    return InMemoryBridge(data=snapshot_data)

@pytest.fixture
def repo(bridge: InMemoryBridge) -> BridgeRepository:
    """Repository wired to test bridge with constant mtime."""
    return BridgeRepository(bridge=bridge, mtime_source=ConstantMtimeSource())

@pytest.fixture
def service(repo: BridgeRepository) -> OperatorService:
    """OperatorService wired to test repo."""
    return OperatorService(repository=repo)
```

Important details:
- The `bridge` fixture reads the `@pytest.mark.snapshot(...)` marker from the test node
- Marker kwargs are passed directly to `make_snapshot_dict(**marker.kwargs)` — this means `@pytest.mark.snapshot(tasks=[...], tags=[...])` just works
- No marker = default snapshot (backward compatible)
- The `service` fixture depends on `repo` which depends on `bridge` — standard pytest composition
- Tests can inject `service` alone (most common), or `service` + `bridge` when they need to inspect bridge calls, or `service` + `repo` when they need to call repo methods directly

async def test_tags_by_name(self) -> None:
