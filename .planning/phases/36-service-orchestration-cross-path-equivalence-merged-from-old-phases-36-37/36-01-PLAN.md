    ```python
    @staticmethod
    def _expand_review_due(f: ReviewDueFilter) -> datetime:
        """Expand ReviewDueFilter to a concrete datetime threshold."""
        now = datetime.now(UTC)
        if f.amount is None:
            return now
        # Map unit to timedelta/relativedelta
        # d -> timedelta(days=amount)

        # m -> now.replace(month=now.month + amount) or use dateutil-free approach
        # y -> now.replace(year=now.year + amount)
    ```
